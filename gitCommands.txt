1) git init - Initialize local git repository
   a) git config --global user.name '<git username>'
   b) git config --global user.email'<git user email>'
2) git add <file> - Add file(s) to staging area
3) git status - Check status of working tree
4) git commit - Commit changes in staging area
5) git push - Push to remote repository
6) git pull - Pull latest from remote repository
7) git clone - Clone remote repository locally
8) git branch <branch name> Create a new branch
9) git checkout <branch name> Switch to a branch
10) git merge <branch name> Merge a branch with another
